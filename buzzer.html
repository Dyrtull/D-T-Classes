// Simple Passive Buzzer Test
const int buzzerPin = 8;

void setup() {
  pinMode(buzzerPin, OUTPUT);
}

void loop() {
  // Play 1000Hz tone
  tone(buzzerPin, 1000); 
  delay(1000);
  
  // Stop the sound
  noTone(buzzerPin);     
  delay(500);
}

  pinMode(ledPin, OUTPUT);
}

  beep(440, 500);

void beep(int note, int duration) {
  // 1. Turn LED ON
  digitalWrite(ledPin, HIGH);
  
  // 2. Play Sound
  tone(buzzerPin, note, duration);
  
  // 3. Wait
  delay(duration);
  
  // 4. Turn LED OFF
  digitalWrite(ledPin, LOW);
  
  delay(50);
}
